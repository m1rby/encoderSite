25char detectMajorityAlphabet(const string& text) {
26 int latinCount = 0;
27 int cyrillicCount = 0;
28
29 for (unsigned char c : text) {
30 if ((c >= 'A' && c <= 'Z') || (c >= 'a' && c <= 'z') || (c >= '0' && c <= '9')) {
31 latinCount++;
32 } else if (c >= 0xC0 && c <= 0xFF) { // кириллические символы в Windows-1251
33 cyrillicCount++;
34 }
35 }
36 return (latinCount > cyrillicCount) ? 'L' : 'C';
37}

208string readFileInParallel(const string& fileName) {
209 ifstream file(fileName, ios::in | ios::binary);
210 if (!file.is_open()) {
211 wcout << L"Произошла ошибка при открытии файла: ";
212 cout << fileName << endl;
213 }
214
215 // Определяем размер файла
216 file.seekg(0, ios::end);
217 size_t fileSize = file.tellg();
218 file.seekg(0, ios::beg);
219
220 // Разделяем чтение на потоки
221 const size_t numThreads = 4; // Количество потоков
222 size_t chunkSize = fileSize / numThreads;
223
224 vector<string> parts(numThreads);
225 vector<thread> threads;
226 mutex mtx;
227
228 for (size_t i = 0; i < numThreads; ++i) {
229 threads.emplace_back([&, i]() {
230 ifstream localFile(fileName, ios::in | ios::binary);
231 localFile.seekg(i * chunkSize, ios::beg);
232
233 // Вычисляем размер части для потока
234 size_t readSize = (i == numThreads - 1) ? (fileSize - i * chunkSize) : chunkSize;
235
236 // Читаем часть файла
237 string localBuffer(readSize, '\0');
238 localFile.read(&localBuffer[0], readSize);
239
240 // Добавляем результат
241 lock_guard<mutex> lock(mtx);
242 parts[i] = move(localBuffer);
243
244 localFile.close();
245 });
246 }
247
248 for (auto& thread : threads) {
249 thread.join();
250 }
251
252 // Объединяем результаты
253 string content;
254 for (auto& part : parts) {
255 content += move(part);
256 }
257
258 file.close();
259
260 return content;
261}

278string encryptFile(const string& text, const string& gamma) {
279 string resultText;
280 const size_t numThreads = 4;
281 size_t chunkSize = text.size() / numThreads;
282
283 vector<string> encryptedParts(numThreads);
284 vector<thread> threads;
285
286 // Параллельное шифрование
287 for (size_t i = 0; i < numThreads; ++i) {
288 threads.emplace_back([&, i]() {
289 size_t start = i * chunkSize;
290 size_t end = (i == numThreads - 1) ? text.size() : start + chunkSize;
291
292 string block = text.substr(start, end - start);
293 encryptedParts[i] = xorBlock(block, gamma); // Используем xorBlock
294 });
295 }
296
297 for (auto& thread : threads) {
298 thread.join();
299 }
300
301 // Объединяем части из вектора
302 for (const auto& part : encryptedParts) {
303 resultText += part;
304 }
305
306 return resultText;
307}

342 mutex writeMutex;

344 // Параллельно записываем данные
345 vector<thread> threads;

348 threads.emplace_back([&, offset](string data, streampos currentOffset) {

353 }
354 outputFile.seekp(currentOffset); // Переходим к нужному смещению
355 outputFile.write(data.data(), data.size());
356 if (!outputFile) {
357 throw runtime_error("Failed to write to file: " + inputFileName);
358 }
359 outputFile.close();
360 }, part, offset);
361 offset += part.size();
362 }

364 // Ожидаем завершения всех потоков
